tree("Root")
	parallel
		repeat mute tree("Shooting")
		repeat
			mute
				while
					sequence
						IsAI
						HasTarget
					sequence
						parallel
							repeat mute tree("Chasing")
							repeat mute tree("Aiming")
							repeat mute tree("IdlingStand")
							repeat mute tree("IdlingAttack")
		repeat
			mute
				while
					sequence
						not IsAI
					sequence
						parallel
							SetMovingInput
							SetAimingInput

tree("IdlingStand")
	repeat
		while
			sequence
				not CanChase
			sequence
				OnIdlingStand

tree("IdlingAttack")
	repeat
		while
			sequence
				CanStopChasing
			sequence
				OnIdlingAttack

tree("Chasing")
	repeat
		while
			sequence
				CanChase
				not CanStopChasing
			sequence
				OnChasing

tree("Shooting")
	repeat
		while
			sequence
				CanShoot
			sequence
				OnShooting

tree("Aiming")
	repeat
		while
			sequence
				CanChase
			sequence
				OnAiming
