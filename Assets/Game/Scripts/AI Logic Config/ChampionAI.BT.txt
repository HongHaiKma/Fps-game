tree("Root")
	parallel
		repeat mute tree("Shooting")
		repeat
			mute
				while
					sequence
						IsAI
					sequence
						parallel
							repeat mute tree("Moving")
							repeat mute tree("Aiming")
							repeat mute tree("Idling")
		repeat
			mute
				while
					sequence
						not IsAI
					sequence
						parallel
							SetMovingInput
							SetAimingInput

tree("Idling")
	repeat
		while
			sequence
				not CanChase
			sequence
				OnIdling

tree("Moving")
	repeat
		while
			sequence
				CanChase
			sequence
				OnChasing

tree("Shooting")
	repeat
		while
			sequence
				CanShoot
			sequence
				OnShooting

tree("Aiming")
	repeat
		while
			sequence
				CanChase
			sequence
				OnAiming
